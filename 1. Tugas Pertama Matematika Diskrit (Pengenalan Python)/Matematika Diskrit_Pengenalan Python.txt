        "def kelulusan_olimpiade():\n",
        "  T = int(input()) # Jumlah seleksi\n",
        "  for i in range(T):\n",
        "    N,M = map(int, input().split()) # Memasukkan jumlah peserta (N) dan jumlah yang lulus (M)\n",
        "    target_id = input().strip() # Mengecek id peserta\n",
        "\n",
        "    peserta = [input().split() for _ in range(N)]\n",
        "\n",
        "    # Sorting peserta\n",
        "    peserta.sort(key=lambda x: (-int(x[3]), -int(x[2]), -int(x[1])))\n",
        "\n",
        "    # Ambil peserta teratas yang lulus\n",
        "    peserta_lulus = {peserta[i][0]: True for i in range(M)}\n",
        "\n",
        "    # Print siapa yang lulus\n",
        "    print(\"YA\" if target_id in peserta_lulus else \"TIDAK\")\n",
        "\n",
